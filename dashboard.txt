  function normalizeDriveFileId(value) {
  if (!value) return '';
  const v = String(value).trim();
  if (!v) return '';

  const patterns = [
    /\/d\/([a-zA-Z0-9_-]{25,})/,
    /[?&]id=([a-zA-Z0-9_-]{25,})/,
    /\/file\/d\/([a-zA-Z0-9_-]{25,})/,
    /\/uc\?.*id=([a-zA-Z0-9_-]{25,})/,
    /^([a-zA-Z0-9_-]{25,})$/
  ];

  for (let i = 0; i < patterns.length; i++) {
    const m = v.match(patterns[i]);
    if (m && m[1]) return m[1];
  }
  return '';
}

function buildImageUrl(raw) {

  return baseUrl + '?action=serveImage&id=' + encodeURIComponent(id);
}

function showAlert(type, message) {
  var alertBox = document.getElementById('alertMessage');
  if (!alertBox) return;

  if (alertBox._hideTimer) {
    clearTimeout(alertBox._hideTimer);
  }
  if (alertBox._removeTimer) {
    clearTimeout(alertBox._removeTimer);
  }

  var rawMessage = String(message || '').trim();
  var safeMessage = rawMessage
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;');

  var icon = 'ℹ️';
  if (type === 'success') icon = '✅';
  if (type === 'error') icon = '❌';
  if (type === 'info') icon = 'ℹ️';

  var matched = rawMessage.match(/โหลดข้อมูลสำเร็จ\s*พบ\s*(\d+)\s*รายการ/i);
  if (matched) {
    var count = matched[1];
    alertBox.innerHTML =
      '<span class="alert-icon">' + icon + '</span>' +
      '<span class="alert-text">' +
      'โหลดข้อมูลสำเร็จ' +
      '<span class="alert-sep">•</span>' +
      'พบ' +
      '<span class="alert-count">' + count + '</span>' +
      'รายการ' +
      '</span>';
  } else {
    alertBox.innerHTML =
      '<span class="alert-icon">' + icon + '</span>' +
      '<span class="alert-text">' + safeMessage + '</span>';
  }

  alertBox.className = 'alert ' + type + ' show';
  alertBox.style.animation = '';

  alertBox._hideTimer = setTimeout(function() {
    var exitAnimation = window.innerWidth <= 768
      ? 'toastSlideOutMobile 0.24s ease forwards'
      : 'toastSlideOutDesktop 0.24s ease forwards';

    alertBox.style.animation = exitAnimation;

    alertBox._removeTimer = setTimeout(function() {
      alertBox.className = 'alert ' + type;
      alertBox.style.animation = '';
      alertBox.innerHTML = '';
    }, 240);
  }, 2600);
}

    function sanitizeTechnicians(list) {
        var seen = {};
        var cleaned = [];

        (Array.isArray(list) ? list : []).forEach(function(name) {
            var safeName = String(name == null ? '' : name)
                .replace(/[\r\n]+/g, ' ')
                .replace(/\s+/g, ' ')
                .trim();

            if (!safeName) return;
            if (seen[safeName]) return;

            seen[safeName] = true;
            cleaned.push(safeName);
        });

        return cleaned;
    }

  function handleDataResponse(result) {
  console.log('📨 Data response received:', result);
  showLoading(false);

  try {
    const processedResult = result;

    if (!processedResult) {
      throw new Error('ไม่ได้รับข้อมูลจากเซิร์ฟเวอร์');
    }

    if (processedResult.hasOwnProperty('success') && !processedResult.success) {
      throw new Error(processedResult.error || 'เซิร์ฟเวอร์ส่งคืนสถานะไม่สำเร็จ');
    }

    if (!processedResult.data || !Array.isArray(processedResult.data)) {
      throw new Error('รูปแบบข้อมูลที่ได้รับไม่ถูกต้อง');
    }

    allData = processedResult.data;
    chartData = processedResult.chartData || null;
    filteredData = allData.slice();
    currentPage = 1;

    console.log('✅ Data loaded successfully:', allData.length, 'items');

    generateTableHeader();
    updateStatistics();

    renderTable();

    deferTask(function() {
      updateCharts();
    });

    if (allData.length > 0) {
      showAlert('success', 'โหลดข้อมูลสำเร็จ พบ ' + allData.length + ' รายการ');
    } else {
      showAlert('info', 'ไม่พบข้อมูลในระบบ Google Sheets');
    }

  } catch (e) {
    console.error('❌ Error processing data:', e);
    handleDataError(e);
  }
}

      function processChartData() {

                                chartData.statusCount.completed++;
                            } else if (item.status === 'ยกเลิก') {
                                chartData.statusCount.cancelled++;

                                case 'ทั่วไป':
                                    chartData.repairTypeCount.general++;

                                case 'ไฟฟ้า':

       function generateTechnicianStats(data) {
  const technicianMap = {};

  const normalizeStatusName = (input) => {
    const s = String(input || '').trim();
    if (s === 'รอเบิกพัสดุ') return 'พัสดุอยู่ระหว่างรอเบิก';
    return s;
  };

  data.forEach(item => {
    const tech = (item.assignedTechnician || '').toString().trim();
    const status = normalizeStatusName(item.status);

    if (!tech || tech === '-') return;

    if (!technicianMap[tech]) {
      technicianMap[tech] = {
        name: tech,
        total: 0,
        pending: 0,
        processing: 0,
        external: 0,
        parts: 0,
        completed: 0,
        cancelled: 0
      };
    }

    technicianMap[tech].total++;

    if (status === 'รอดำเนินการ') technicianMap[tech].pending++;
    else if (status === 'กำลังดำเนินการ') technicianMap[tech].processing++;
    else if (status === 'ดำเนินการภายนอก') technicianMap[tech].external++;
    else if (status === 'พัสดุอยู่ระหว่างรอเบิก') technicianMap[tech].parts++;
    else if (status === 'เสร็จสิ้น') technicianMap[tech].completed++;
    else if (status === 'ยกเลิก') technicianMap[tech].cancelled++;
  });

  return Object.values(technicianMap)
    .sort((a, b) => b.total - a.total)
    .slice(0, 10);
}

// [ANCHOR: DASHBOARD-LOW-STOCK-ALERT-WITH-SUMMARY-SEVERITY]
function updateLowStockAlert() {
  var lowStockCountEl = document.getElementById('lowStockCount');
  var lowStockListEl = document.getElementById('lowStockList');

  if (!lowStockCountEl || !lowStockListEl) return;

  var items = [];
  if (chartData && chartData.lowStockItems && chartData.lowStockItems.length > 0) {
    items = chartData.lowStockItems;
  }

  function getSeverity(item) {
    var balance = Number(item.balance || 0);
    var reorderPoint = Number(item.reorderPoint || 0);

    if (balance <= 0) {
      return {
        label: 'หมดสต็อก',
        badgeBg: '#dc2626',
        itemBg: 'linear-gradient(135deg, #fef2f2 0%, #fff1f2 100%)',
        borderColor: '#fca5a5',
        textColor: '#991b1b',
        level: 3
      };
    }

    if (balance < reorderPoint) {
      return {
        label: 'ต่ำกว่าจุดเตือน',
        badgeBg: '#f97316',
        itemBg: 'linear-gradient(135deg, #fff7ed 0%, #fffbeb 100%)',
        borderColor: '#fdba74',
        textColor: '#9a3412',
        level: 2
      };
    }

    if (balance === reorderPoint) {
      return {
        label: 'ถึงจุดเตือน',
        badgeBg: '#eab308',
        itemBg: 'linear-gradient(135deg, #fefce8 0%, #fffbeb 100%)',
        borderColor: '#fde047',
        textColor: '#854d0e',
        level: 1
      };
    }

    return {
      label: 'ปกติ',
      badgeBg: '#10b981',
      itemBg: 'linear-gradient(135deg, #ecfdf5 0%, #f0fdf4 100%)',
      borderColor: '#86efac',
      textColor: '#166534',
      level: 0
    };
  }


  function getSummarySeverity(list) {
    var maxLevel = 0;

    list.forEach(function(item) {
      var severity = getSeverity(item);
      if (severity.level > maxLevel) {
        maxLevel = severity.level;
      }
    });

    if (maxLevel === 3) {
      return {
        bg: 'linear-gradient(135deg, #fef2f2 0%, #fff1f2 100%)',
        border: '#fca5a5',
        countColor: '#dc2626',
        text: 'ต้องรีบเติมสต็อก'
      };
    }

    if (maxLevel === 2) {
      return {
        bg: 'linear-gradient(135deg, #fff7ed 0%, #fffbeb 100%)',
        border: '#fdba74',
        countColor: '#f97316',
        text: 'ต่ำกว่าจุดเตือน'
      };
    }

    if (maxLevel === 1) {
      return {
        bg: 'linear-gradient(135deg, #fefce8 0%, #fffbeb 100%)',
        border: '#fde047',
        countColor: '#ca8a04',
        text: 'ถึงจุดเตือน'
      };
    }

    return {
      bg: 'linear-gradient(135deg, #ecfdf5 0%, #f0fdf4 100%)',
      border: '#86efac',
      countColor: '#059669',
      text: 'สถานะปกติ'
    };
  }

  lowStockCountEl.textContent = items.length;

  var metricCard = lowStockCountEl.closest('.low-stock-metric');
  var metricText = metricCard ? metricCard.querySelector('span') : null;

  if (items.length === 0) {
    if (metricCard) {
      metricCard.style.background = 'linear-gradient(135deg, #ecfdf5 0%, #f0fdf4 100%)';
      metricCard.style.borderColor = '#86efac';
    }
    lowStockCountEl.style.color = '#059669';
    if (metricText) metricText.textContent = 'สถานะปกติ';

    lowStockListEl.innerHTML = '<div class="low-stock-empty">ยังไม่มีรายการอะไหล่ใกล้หมด</div>';
    return;
  }

  var summarySeverity = getSummarySeverity(items);

  if (metricCard) {
    metricCard.style.background = summarySeverity.bg;
    metricCard.style.borderColor = summarySeverity.border;
  }
  lowStockCountEl.style.color = summarySeverity.countColor;
  if (metricText) metricText.textContent = summarySeverity.text;

  lowStockListEl.innerHTML = items.map(function(item) {
    var severity = getSeverity(item);

    return `
      <div class="low-stock-item" style="background:${severity.itemBg}; border-color:${severity.borderColor};">
        <div class="low-stock-item-head">
          <div class="low-stock-name" style="color:${severity.textColor};">${item.fullName || item.name}</div>
          <div class="low-stock-badge" style="background:${severity.badgeBg};">${severity.label}</div>
        </div>
        <div class="low-stock-meta">
          รหัส: ${item.partId}<br>
          คงเหลือ: ${item.balance} ชิ้น<br>
          จุดเตือน: ${item.reorderPoint}
        </div>
      </div>
    `;
  }).join('');

  console.log('✅ Low stock alert updated successfully');
}

function beginBusy(title, sub) {
  __glCount++;
  setGlobalLoading(true, title || 'กำลังประมวลผล', sub || 'โปรดรอสักครู่');
}

function endBusy() {
  __glCount = Math.max(0, __glCount - 1);
  if (__glCount === 0) setGlobalLoading(false);
}

function runServer(methodName, args, title, sub, onSuccess, onFail) {
  beginBusy(title, sub);

  try {
    if (typeof google === 'undefined' || !google.script || !google.script.run) {
      endBusy();
      if (onFail) onFail(new Error('Google script not available'));
      return;
    }

    let chain = google.script.run
      .withSuccessHandler(function (res) {
        endBusy();
        if (onSuccess) onSuccess(res);
      })
      .withFailureHandler(function (err) {
        endBusy();
        if (onFail) onFail(err);
      });

    if (Array.isArray(args) && args.length) chain[methodName].apply(chain, args);
    else chain[methodName]();

  } catch (e) {
    endBusy();
    if (onFail) onFail(e);
  }
}

          function correctDate(dateStr) {
    if (!dateStr || !dateStr.includes('/')) return dateStr;
    const parts = dateStr.split('/');
    if (parts.length !== 3) return dateStr;
    let year = parseInt(parts[2]);
    if (year > 2600) {
        year -= 543;
        if (year < 1900 || year > 2100) year = new Date().getFullYear() + 543;
    } else if (year < 2400) {
        year += 543;
    }
    return `${parts[0]}/${parts[1]}/${year}`;
}

  if (!tableBody || !dataTable || !emptyState) return;

  if (!filteredData || filteredData.length === 0) {
    dataTable.style.display = 'none';
    if (paginationControls) paginationControls.style.display = 'none';
    emptyState.style.display = 'block';
    tableBody.innerHTML = '';
    return;
  }

  dataTable.style.display = 'table';
  emptyState.style.display = 'none';

  const totalRows = filteredData.length;
  const totalPages = Math.ceil(totalRows / rowsPerPage);

  if (currentPage < 1) currentPage = 1;
  if (currentPage > totalPages) currentPage = totalPages;

  const startIndex = (currentPage - 1) * rowsPerPage;
  const endIndex = Math.min(startIndex + rowsPerPage, totalRows);
  const currentData = filteredData.slice(startIndex, endIndex);

  tableBody.innerHTML = '';

  const myToken = ++tableRenderToken;

  function appendRowsInBatch(items, start) {
    if (myToken !== tableRenderToken) return;

    const fragment = document.createDocumentFragment();
    const batchSize = isMobileViewport() ? 4 : items.length;
    const end = Math.min(start + batchSize, items.length);

    for (let i = start; i < end; i++) {
      const row = createTableRow(items[i]);
      fragment.appendChild(row);
    }

    tableBody.appendChild(fragment);

    if (end < items.length) {
      nextFrame(function() {
        appendRowsInBatch(items, end);
      });
    }
  }

  appendRowsInBatch(currentData, 0);

  if (paginationControls) {
    paginationControls.style.display = 'flex';

    const pageInfo = document.getElementById('pageInfo');
    const prevBtn = document.getElementById('prevPageBtn');
    const nextBtn = document.getElementById('nextPageBtn');

    if (pageInfo) {
      pageInfo.textContent = 'แสดง ' + (startIndex + 1) + ' - ' + endIndex + ' จาก ' + totalRows + ' รายการ';
    }

    if (prevBtn) prevBtn.disabled = currentPage <= 1;
    if (nextBtn) nextBtn.disabled = currentPage >= totalPages;
  }
}

            renderTable(); // เรนเดอร์ใหม่เฉพาะส่วนข้อมูล

function createTableRow(item) {
    const row = document.createElement('tr');
    row.className = 'fade-in';
    const sequenceEscaped = (item.sequenceNumber || '').replace(/'/g, "\\'");
    
    // ตรวจสอบข้อมูลรูปภาพ
    const validImageUrls = (item.imageUrls && Array.isArray(item.imageUrls)) ?
        item.imageUrls.filter(url => url && typeof url === 'string' && url.trim() !== '') : [];
    const hasImages = validImageUrls.length > 0;
    
    // สร้างปุ่ม Actions HTML เต็มรูปแบบ
    let actionsHTML = '';

    // ปุ่มดูเอกสาร (PDF)
    if (item.url && item.url !== '#' && item.url.trim()) { 
        actionsHTML += `<a href="${item.url}" target="_blank" class="action-btn view" title="ดูเอกสาร"><i class="fas fa-file-alt action-icon icon-view"></i> ดู</a>`; 
    }
    
    // ปุ่มดูรูปภาพ
    if (hasImages) { 
        actionsHTML += `<button class="action-btn images" onclick="showImageGallery('${sequenceEscaped}')" title="ดูรูปภาพ (${validImageUrls.length} รูป)"><i class="fas fa-camera-retro action-icon icon-images"></i> รูป (${validImageUrls.length})</button>`; 
    } else { 
        actionsHTML += `<button class="action-btn images" disabled title="ไม่มีรูปภาพประกอบ"><i class="fas fa-camera-slash action-icon icon-images"></i> ไม่มีรูป</button>`; 
    }
    
    // ปุ่มรายละเอียด (แสดงตลอด)
    actionsHTML += `<button class="action-btn info" onclick="showDetail('${sequenceEscaped}')" title="ดูรายละเอียด"><i class="fas fa-eye action-icon icon-detail"></i> รายละเอียด</button>`;
    
    // ปุ่มสำหรับ Admin/Technician (แก้ไข/ลบ)

            actionsHTML += `<button class="action-btn edit" onclick="updateStatus('${sequenceEscaped}')" title="แก้ไขสถานะ"><i class="fas fa-wrench action-icon icon-edit"></i> แก้ไข</button>`;

        }
        // ปุ่มมอบหมายงาน (เฉพาะ Admin)
        if (currentUser.role === 'admin') {
            actionsHTML += `<button class="action-btn assign" onclick="assignTask('${sequenceEscaped}')" title="มอบหมายงานให้ช่าง"><i class="fas fa-user-check action-icon icon-assign"></i> มอบหมาย</button>`;
        }
    }
    
    const statusClass = getStatusClass(item.status);
    
    // CHANGE: item.date ตอนนี้เป็น String "dd/MM/yyyy HH:mm น." จาก Server แล้ว นำมาแสดงได้เลย
    row.innerHTML = `
        <td data-label="เลขที่:"><div class="table-row-number"><i class="fas fa-hashtag"></i>${item.sequenceNumber || '-'}</div></td>
        <td data-label="วันที่/เวลา:">
            <div class="table-date" style="flex-direction: column; align-items: flex-start; gap: 2px;">
                <div style="display:flex; align-items:center; gap:5px;">
                    <i class="fas fa-calendar-alt"></i>
                    <span style="font-weight:600; font-size: 0.9em;">${item.date || '-'}</span>
                </div>
            </div>
        </td>
        <td data-label="ผู้แจ้งซ่อม:"><div class="table-requester"><i class="fas fa-user"></i><span>${item.requesterName || '-'}</span></div></td>
        <td data-label="รายการ:"><div class="table-repair-item"><i class="fas fa-tools"></i><div>${item.repairItem || '-'}</div></div></td>
        <td data-label="สถานที่:"><div class="table-location"><i class="fas fa-map-marker-alt"></i><div>${item.location || '-'}</div></div></td>
        <td data-label="ผู้รับผิดชอบ:"><div class="table-technician"><i class="fas fa-user-cog"></i><span>${item.assignedTechnician || '-'}</span></div></td>

          function handleEditImageSelect(input) {
            const files = Array.from(input.files);
            if (files.length === 0) return;

            // ตรวจสอบจำนวนรูป
            if (editSelectedImages.length + files.length > 3) {
                showAlert('error', 'อัปโหลดได้สูงสุด 3 รูปเท่านั้นค่ะ');
                return;
            }

            files.forEach(file => {
                if (file.size > 5 * 1024 * 1024) { // 5MB limit
                    showAlert('error', `ไฟล์ ${file.name} ใหญ่เกินไป (max 5MB)`);
                    return;
                }

                const reader = new FileReader();
                reader.onload = (e) => {
                    editSelectedImages.push({
                        file: file,
                        dataURL: e.target.result
                    });
                    renderEditImagePreview();
                };
                reader.readAsDataURL(file);
            });
            input.value = ''; // Reset input
        }

// CHANGE: รวมให้เหลือ 1 ชุดเท่านั้น (กันฟังก์ชันซ้ำทับกัน) + เปิด PDF จาก result.pdfUrl + ตัด demo/mock ออก

function ensureBuddhistYear(year) {
  // [ANCHOR: REPORT-YEAR-NORMALIZE]
  // CHANGE: รวม logic แปลงปีไว้จุดเดียว
  var y = (typeof year === 'number') ? year : parseInt(year, 10);

  if (!isFinite(y)) return (new Date().getFullYear() + 543);

  // ถ้าเป็น ค.ศ. ให้แปลงเป็น พ.ศ.
  if (y < 2400) return y + 543;

  // ถ้าเลขปีเพี้ยนเกินจริง ให้พยายามแก้กลับ (กรณีบวก 543 ซ้ำ)
  if (y > 3000) {
    // ลดทีละ 543 จนเข้าโซนสมเหตุสมผล
    while (y > 3000) y -= 543;
    if (y < 2400) y += 543;
  }

  return y;
}

function generateMonthlyReport() {
  console.log('🗓️ Starting monthly report generation...');

  var selector = document.getElementById('monthSelector');
  var technicianSelector = document.getElementById('reportTechnicianSelector');
  var generateBtn = document.getElementById('generateReportBtn');

  function normalizeTechnicianName(value) {
    return String(value == null ? '' : value)
      .replace(/[\r\n]+/g, ' ')
      .replace(/\s+/g, ' ')
      .trim();
  }

  function resetGenerateButton() {
    if (!generateBtn) return;
    generateBtn.disabled = false;
    generateBtn.removeAttribute('aria-busy');
    generateBtn.innerHTML = '<i class="fas fa-file-alt"></i> สร้างรายงาน';
  }

  console.log('🔍 monthSelector found:', !!selector);
  console.log('🔍 reportTechnicianSelector found:', !!technicianSelector);
  console.log('🔍 generateReportBtn found:', !!generateBtn);

  if (!selector || !selector.value) {
    showAlert('error', 'กรุณาเลือกเดือนที่ต้องการสร้างรายงาน');
    return;
  }

  var selectedTechnician = '';
  if (technicianSelector && !technicianSelector.disabled) {
    selectedTechnician = normalizeTechnicianName(technicianSelector.value || '');
  }

  console.log('📅 Selected month dropdown:', selector.value);
  console.log('👨‍🔧 Selected technician dropdown:', selectedTechnician || 'รวมทุกคน');
  console.log('📌 Technician selector disabled state:', technicianSelector ? technicianSelector.disabled : 'not-found');

  var parts = selector.value.split('-');
  if (parts.length !== 2) {
    showAlert('error', 'รูปแบบข้อมูลเดือนไม่ถูกต้อง');
    return;
  }

  var yearRaw = parseInt(parts[0], 10);
  var month = parseInt(parts[1], 10);

  if (!isFinite(month) || month < 1 || month > 12) {
    showAlert('error', 'เดือนไม่ถูกต้อง');
    return;
  }

  var year = ensureBuddhistYear(yearRaw);
  if (generateBtn && generateBtn.disabled) return;

  if (generateBtn) {
    generateBtn.disabled = true;
    generateBtn.setAttribute('aria-busy', 'true');
    generateBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> กำลังสร้าง...';
  }

  var monthName = (typeof MONTHS_TH !== 'undefined' && MONTHS_TH[month - 1]) ? MONTHS_TH[month - 1] : ('เดือน ' + month);
  var reportLabel = selectedTechnician
    ? ('รายงานของ ' + selectedTechnician + ' เดือน ' + monthName + ' ' + year)
    : ('รายงานรวมทุกคน เดือน ' + monthName + ' ' + year);

  showAlert('info', 'กำลังสร้าง' + reportLabel + ' กรุณารอสักครู่...');

  console.log('📦 Report payload before send:', {
    year: year,
    month: month,
    technician: selectedTechnician || '',
    mode: selectedTechnician ? 'single-technician' : 'all-technicians'
  });
  console.log('🔘 Generate button state before request:', generateBtn ? generateBtn.disabled : 'no-button');

  try {
    if (!(typeof google !== 'undefined' && google.script && google.script.run)) {
      throw new Error('Google Apps Script ไม่พร้อมใช้งาน');
    }

    google.script.run
      .withSuccessHandler(function(result) {
        resetGenerateButton();

        console.log('✅ Monthly report result:', result);
        console.log('🔘 Generate button state after success:', generateBtn ? generateBtn.disabled : 'no-button');

        if (result && result.success) {
          if (result.isEmpty) {
            console.log('📭 Empty report result for technician:', selectedTechnician || 'รวมทุกคน');
            showAlert('info', result.message || 'ไม่พบรายการแจ้งซ่อมในเงื่อนไขที่เลือก');

            if (result.reportData) {
              result.reportData.year = year;
              showReportPreview(result.reportData, true, null);
            }
            return;
          }

          showAlert('success', '✅ สร้างรายงาน "' + (result.docName || '') + '" เรียบร้อยแล้ว');

            window.open(result.pdfUrl, '_blank');
          } else {
            showAlert('error', 'สร้างรายงานสำเร็จ แต่ไม่พบลิงก์ PDF');
          }

          if (result.reportData) {
            result.reportData.year = year;
            showReportPreview(result.reportData, false, result.pdfUrl);
          }
        } else {
          var errorMsg = (result && result.error) ? result.error : 'ไม่ทราบสาเหตุ';
          showAlert('error', '❌ เกิดข้อผิดพลาดในการสร้างรายงาน: ' + errorMsg);

        resetGenerateButton();

        console.error('❌ Monthly report server error:', error);
        console.log('🔘 Generate button state after failure:', generateBtn ? generateBtn.disabled : 'no-button');

        var msg = (error && error.message) ? error.message : String(error || '');

  } catch (error) {
    resetGenerateButton();

    console.error('❌ Monthly report client error:', error);
    console.log('🔘 Generate button state after catch:', generateBtn ? generateBtn.disabled : 'no-button');

    showAlert('error', '❌ เกิดข้อผิดพลาด: ' + (error && error.message ? error.message : String(error)));
  }

        // ===== เริ่มรัน Log =====
        terminal.innerHTML = `<span style="color:#60a5fa; font-weight:bold;">[${new Date().toLocaleTimeString()}] 🍓 เริ่มต้นการตรวจสอบระบบ Full Lifecycle...</span><br><br>`;
        
        if (btn) btn.disabled = true;
        if (btnCopy) btnCopy.disabled = true; // ล็อกปุ่ม Copy ตอนรัน

        const addLog = (step, isPass, msg) => {
            const icon = isPass ? "✅ PASS" : "❌ FAIL";
            const color = isPass ? "#10b981" : "#ef4444";
            terminal.innerHTML += `<span style="color:${color}; font-weight:500;">${icon} | ${step}: ${msg}</span><br>`;
            terminal.scrollTop = terminal.scrollHeight; // Auto Scroll ไปล่างสุด
        };

        try {
            // 1. Client Check
            addLog("Client UI", true, "ตรวจสอบความพร้อมหน้าจอแดชบอร์ด");

            // 2. Chart Check
            const mandatoryStatuses = ["รอดำเนินการ", "กำลังดำเนินการ", "รอเบิกพัสดุ", "เสร็จสิ้น", "ยกเลิก"];
            if (typeof statusChart !== 'undefined' && statusChart?.data?.labels) {
                const currentLabels = statusChart.data.labels;
                const hasAll = mandatoryStatuses.every(s => currentLabels.includes(s));
                addLog("Chart Mapping", hasAll, hasAll ? "รองรับครบ 5 สถานะหลัก" : "สถานะในกราฟไม่ครบ");
            } else {
                addLog("Chart Engine", false, "ไม่พบข้อมูลกราฟ");
            }

            // 3. Server Check
            addLog("Server Connection", true, "กำลังส่งคำสั่งไปที่ Google Apps Script...");
            terminal.innerHTML += `<span style="color:#94a3b8; font-style:italic;">...รอสักครู่นะคะ กำลังคุยกับหลังบ้าน...</span><br>`;

            google.script.run
                .withSuccessHandler(function(res) {
                    if (res && res.logs) {
                        res.logs.forEach(l => {
                            // แปลง log server เป็น html
                            const cleanLog = String(l).replace('PASS', '').replace('FAIL', '').trim();
                            const isPass = String(l).includes("✅");
                            addLog("Server Result", isPass, cleanLog);
                        });
                        terminal.innerHTML += `<br><span style="color:#fbbf24; font-weight:bold;">✨ การทดสอบเสร็จสมบูรณ์ 100% แล้วค่ะ!</span>`;
                    } else {
                        addLog("Server Error", false, "ได้ข้อมูลเปล่ากลับมา");
                    }
                    finishTest();
                })
                .withFailureHandler(function(err) {
                    addLog("Fatal Error", false, err.message || err);
                    finishTest();
                })
                .selfTest(); // เรียกฟังก์ชัน selfTest() ใน code.gs

        } catch (e) {
            addLog("Client Error", false, e.message);
            finishTest();
        }

        function finishTest() {
            window.selfTestRunning = false;
            if (btn) btn.disabled = false;
            if (btnCopy) btnCopy.disabled = false; // ปลดล็อกปุ่ม Copy
        }
    }

    function closeSelfTestModal() {
        console.log("🍒 Action: Closing SelfTest Modal...");
        
        const modal = document.getElementById('selfTestModal');
        const btn = document.getElementById('btnTestSystem');
        const btnCopy = document.getElementById('btnCopyTerminal');

        if (!modal) {
            console.warn("Modal element not found!");
            return;
        }

        // 1. FORCE RESET STATE (สำคัญมาก: ปลดล็อกสถานะการรันทันที)
        window.selfTestRunning = false;

        // 2. UI CLEANUP (ปิด Modal ทันที)
        modal.classList.remove('active');
        // รอ Animation จบนิดนึงค่อยซ่อน หรือซ่อนเลยถ้าจำเป็น
        setTimeout(() => {
            modal.style.display = 'none'; 
            modal.setAttribute('aria-hidden', 'true');
        }, 100); // delay เล็กน้อยเพื่อให้ effect zoomOut ทำงาน (ถ้ามี) หรือ set 0 ถ้าต้องการทันที

        // 3. RESTORE SCROLL (คืนค่า Scroll ให้ Body)
        try {
            const y = parseInt(document.body.dataset.scrollY || '0');
            document.body.style.position = '';
            document.body.style.top = '';
            document.body.style.width = '';
            document.body.style.overflow = ''; // เผื่อมี overflow hidden ค้าง
            window.scrollTo(0, y);
        } catch (e) {
            console.error("Scroll restore error (non-fatal):", e);
        }

        // 4. BUTTON RESET (คืนค่าปุ่มต่างๆ)
        if (btn) btn.disabled = false;
        if (btnCopy) btnCopy.disabled = false;
        
        // 5. CLEAR LOG (Optional: ถ้าต้องการเคลียร์ Log เมื่อปิด)
        // document.getElementById('terminalLog').innerHTML = '[Ready] รอการเริ่มทดสอบ...';
        
        // Return Focus
        if (btn) btn.focus();
    }

    function copyTerminalLog() {
        const terminal = document.getElementById('terminalLog');
        const btnCopy = document.getElementById('btnCopyTerminal');
        if (!terminal) return;

        // กันกดซ้ำ
        if(btnCopy) btnCopy.disabled = true;

        const textToCopy = terminal.innerText.trim();
        const originalText = btnCopy.innerHTML;

        const success = () => {
            btnCopy.innerHTML = '<i class="fas fa-check"></i> เรียบร้อย!';
            if (typeof showAlert === 'function') showAlert('success', 'คัดลอก Log แล้วค่ะ');
            setTimeout(() => { 
                btnCopy.innerHTML = originalText; 
                btnCopy.disabled = false;
            }, 2000);
        };

        const fail = (err) => {
            console.error("Copy failed", err);
            // Fallback: Textarea Hack
            try {
                const ta = document.createElement('textarea');
                ta.value = textToCopy;
                ta.style.position = 'fixed'; 
                ta.style.left = '-9999px';
                document.body.appendChild(ta);
                ta.select();
                document.execCommand('copy');
                document.body.removeChild(ta);
                success();
            } catch (e) {
                btnCopy.innerHTML = '<i class="fas fa-times"></i> ไม่ได้ค่ะ';
                setTimeout(() => { 
                    btnCopy.innerHTML = originalText;
                    btnCopy.disabled = false; 
                }, 2000);
            }
        };

        if (navigator.clipboard && navigator.clipboard.writeText) {
            navigator.clipboard.writeText(textToCopy).then(success).catch(fail);
        } else {
            fail("API not supported");
        }
    }
