        function initializeSystem() {
            console.log('Initializing repair system...');
            try {
                setCurrentDate();
                generateTemporarySequenceNumber();
                initializeImageUpload();
                initializeEventListeners();
                checkLoginStatus();
                console.log('System initialized successfully');
            } catch (error) {
                console.error('Error initializing system:', error);
                showAlert('error', 'เกิดข้อผิดพลาดในการเริ่มต้นระบบ: ' + error.message);
            }
        }

        function initializeEventListeners() {
            const imageInput = document.getElementById('imageInput');
            const uploadArea = document.querySelector('.image-upload-area');
            const locationSelect = document.getElementById('locationSelect');
            const form = document.getElementById('repairForm');

            if (imageInput) imageInput.addEventListener('change', handleImageSelect);
            if (uploadArea) {
                uploadArea.addEventListener('dragover', handleDragOver);
                uploadArea.addEventListener('drop', handleDrop);
                uploadArea.addEventListener('dragenter', handleDragEnter);
                uploadArea.addEventListener('dragleave', handleDragLeave);
            }
            if (locationSelect) locationSelect.addEventListener('change', handleLocationChange);
            if (form) form.addEventListener('submit', handleFormSubmit);
            console.log('Event listeners initialized');
        }

        function checkLoginStatus() {
            try {
                const loginBtn = document.getElementById('loginBtn');
                const logoutBtn = document.getElementById('logoutBtn');
                const userInfoString = localStorage.getItem('userInfo');

                if (loginBtn && logoutBtn) {
                    if (userInfoString) {
                        const user = JSON.parse(userInfoString);
                        if (user && user.username) {
                            userSession = user;
                            loginBtn.classList.add('hidden');
                            logoutBtn.classList.remove('hidden');
                            console.log('User is logged in:', user.username);
                            return;
                        }
                    }
                    loginBtn.classList.remove('hidden');
                    logoutBtn.classList.add('hidden');
                }
            } catch (error) {
                console.error('Error checking login status:', error);
                localStorage.removeItem('userInfo');
                const loginBtn = document.getElementById('loginBtn');
                const logoutBtn = document.getElementById('logoutBtn');
                if (loginBtn && logoutBtn) {
                    loginBtn.classList.remove('hidden');
                    logoutBtn.classList.add('hidden');
                }
            }
        }

         function setCurrentDate() {
            const dateInput = document.getElementById('currentDate');
            if (dateInput) {

        function processSelectedFiles(files) {
            const imageFiles = files.filter(file => file.type.startsWith('image/'));
            if (imageFiles.length === 0) {
                showAlert('error', 'กรุณาเลือกไฟล์รูปภาพเท่านั้น');
                return;
            }

            const availableSlots = MAX_IMAGES - selectedImages.length;
            if (availableSlots <= 0) {
                showAlert('error', `สามารถแนบรูปภาพได้สูงสุด ${MAX_IMAGES} รูปเท่านั้น`);
                return;
            }

            const filesToProcess = imageFiles.slice(0, availableSlots);
            filesToProcess.forEach(file => {
                if (file.size > MAX_FILE_SIZE) {
                    showAlert('error', `ไฟล์ ${file.name} มีขนาดใหญ่เกิน 5MB`);
                    return;
                }
                addImageToPreview(file);
            });
            if (imageFiles.length > availableSlots) {
                showAlert('info', `เลือกรูปภาพได้อีก ${availableSlots} รูปเท่านั้น (ระบบได้เพิ่มรูปภาพ ${filesToProcess.length} รูปแรกให้แล้ว)`);
            }
        }

        const MOBILE_MAX_IMAGE_WIDTH = 1280;
const MOBILE_IMAGE_QUALITY = 0.82;

function isMobileDevice() {
  return window.matchMedia('(max-width: 768px)').matches;
}

function readFileAsDataURL(file) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onload = function(e) {
      resolve(e.target.result);
    };
    reader.onerror = function(err) {
      reject(err || new Error('ไม่สามารถอ่านไฟล์รูปภาพได้'));
    };
    reader.readAsDataURL(file);
  });
}

function loadImageElement(dataUrl) {
  return new Promise((resolve, reject) => {
    const img = new Image();
    img.onload = function() {
      resolve(img);
    };
    img.onerror = function() {
      reject(new Error('ไม่สามารถโหลดรูปภาพเพื่อย่อขนาดได้'));
    };
    img.src = dataUrl;
  });
}

async function compressImageFile(file) {
  const rawDataUrl = await readFileAsDataURL(file);
  const img = await loadImageElement(rawDataUrl);

  const canvas = document.createElement('canvas');
  const ctx = canvas.getContext('2d', { alpha: false });

  let width = img.width;
  let height = img.height;

  const maxWidth = isMobileDevice() ? MOBILE_MAX_IMAGE_WIDTH : 1600;

  if (width > maxWidth) {
    height = Math.round((height * maxWidth) / width);
    width = maxWidth;
  }

  canvas.width = width;
  canvas.height = height;

  ctx.drawImage(img, 0, 0, width, height);

  const outputType = 'image/jpeg';
  const compressedDataUrl = canvas.toDataURL(outputType, MOBILE_IMAGE_QUALITY);

  return {
    id: Date.now() + Math.random(),
    file: file,
    dataURL: compressedDataUrl,
    originalName: file.name,
    originalSize: file.size,
    compressedSize: Math.round((compressedDataUrl.length * 3) / 4),
    width: width,
    height: height
  };
}

       async function handleFormSubmit(event) {
  event.preventDefault();
  console.log('[STEP1] handleFormSubmit start');

  const locationSelect = document.getElementById('locationSelect');
  const otherLocation = document.getElementById('otherLocation');
  const locationDetail = document.getElementById('locationDetail');

  let finalLocation = '';

  if (locationSelect.value === 'อื่นๆ') {
    if (!otherLocation.value.trim()) {
      showAlert('error', 'กรุณาระบุสถานที่อื่นๆ');
      return;
    }
    finalLocation = otherLocation.value.trim();
  } else if (locationSelect.value) {
    finalLocation = locationSelect.value;
  } else {
    showAlert('error', 'กรุณาเลือกสถานที่');
    return;
  }

  if (locationDetail.value.trim()) {
    finalLocation += ' (' + locationDetail.value.trim() + ')';
  }

  const formData = {
    requesterName: document.getElementById('requesterName').value.trim(),
    phoneNumber: document.getElementById('phoneNumber').value.trim(),
    repairType: document.getElementById('repairType').value,
    priority: document.getElementById('priority').value,
    repairItem: document.getElementById('repairItem').value.trim(),
    description: document.getElementById('description').value.trim(),
    location: finalLocation
  };

  if (!formData.requesterName || !formData.repairItem || !formData.description || !formData.location) {
    showAlert('error', 'กรุณากรอกข้อมูลที่จำเป็นให้ครบถ้วน');
    return;
  }

  toggleLoading(true);
  if (submitBtn) submitBtn.disabled = true;

  try {
    console.log('[STEP2] preparing images:', selectedImages.length);

    const base64Images = selectedImages.map(img => img.dataURL);

    google.script.run
      .withSuccessHandler(function(result) {
        console.log('[STEP3] submitRepairForm success:', result);

        toggleLoading(false);
        if (submitBtn) submitBtn.disabled = false;

        if (!result || !result.success) {
          showAlert('error', (result && result.error) ? result.error : 'บันทึกข้อมูลไม่สำเร็จ');
          return;

        selectedImages = [];

        updateImageCount();
        handleLocationChange();

        window.scrollTo({ top: 0, behavior: 'smooth' });
      })

        toggleLoading(false);
        if (submitBtn) submitBtn.disabled = false;

        showAlert('error', 'เกิดข้อผิดพลาดในการส่งข้อมูล: ' + (error.message || error));
      })

    toggleLoading(false);
    if (submitBtn) submitBtn.disabled = false;
    showAlert('error', 'เกิดข้อผิดพลาด: ' + (error.message || error));
  }
}

    
function handleSubmitSuccess(result) {
  toggleLoading(false);

  if (!result || !result.success) {
    showAlert('error', `❌ เกิดข้อผิดพลาด: ${(result && result.error) ? result.error : 'ไม่ทราบสาเหตุ'}`);
    return;
  }

  const seq = result.sequenceNumber || '';
  const successMessage =
    `✅ ส่งแจ้งซ่อมสำเร็จแล้ว!<br>` +
    `📋 เลขที่: <strong>${seq || 'N/A'}</strong><br>` +
    `🔍 ระบบกำลังสร้างเอกสาร PDF ในเบื้องหลัง`;
  showAlert('success', successMessage);

  const hasRun = (typeof google !== 'undefined' && google.script && google.script.run);

  if (hasRun && seq) {
    try {
      google.script.run
        .withSuccessHandler(function () {
          console.log(`Enqueued PDF job for ${seq}`);
        })
        .withFailureHandler(function (err) {
          console.error('Failed to enqueue PDF job:', err);
        })
        .enqueuePdf(seq);
    } catch (e) {
      console.error('enqueuePdf call failed:', e);
    }

    try {
      google.script.run
        .withSuccessHandler(function () {
          console.log('PDF worker process initiated.');
        })
        .withFailureHandler(function () {})
        .pdfWorker(1);
    } catch (e) {
      console.error('pdfWorker call failed:', e);
    }
  } else {
    if (!hasRun) console.warn('google.script.run not available, skipping PDF background tasks.');
    if (!seq) console.warn('No sequenceNumber returned, skipping PDF background tasks.');
  }

  const form = document.getElementById('repairForm');
  if (form) form.reset();

  if (typeof handleLocationChange === 'function') handleLocationChange();
  if (typeof generateTemporarySequenceNumber === 'function') generateTemporarySequenceNumber();

  selectedImages = [];
  if (typeof renderImagePreview === 'function') renderImagePreview();
  if (typeof updateImageCount === 'function') updateImageCount();
}

        let __glCount = 0;

function setGlobalLoading(show, title, sub) {
  const ov = document.getElementById('globalLoadingOverlay');
  const t = document.getElementById('glTitle');
  const s = document.getElementById('glSub');
  if (!ov) return;

  if (title && t) t.textContent = title;

function beginBusy(title, sub) {
  __glCount++;
  setGlobalLoading(true, title || 'กำลังประมวลผล', sub || 'โปรดรอสักครู่');
}

function endBusy() {
  __glCount = Math.max(0, __glCount - 1);
  if (__glCount === 0) setGlobalLoading(false);
}

function runServer(methodName, args, title, sub, onSuccess, onFail) {
  beginBusy(title, sub);

  try {
    if (typeof google === 'undefined' || !google.script || !google.script.run) {
      endBusy();
      if (onFail) onFail(new Error('Google script not available'));
      return;
    }

    let chain = google.script.run
      .withSuccessHandler(function (res) {
        endBusy();
        if (onSuccess) onSuccess(res);
      })
      .withFailureHandler(function (err) {
        endBusy();
        if (onFail) onFail(err);
      });

    if (Array.isArray(args) && args.length) chain[methodName].apply(chain, args);
    else chain[methodName]();

  } catch (e) {
    endBusy();
    if (onFail) onFail(e);
  }
}

  function loadReports() {
  if (repState.loading) return;

  repState.loading = true;
  updateReportsUI();

  beginBusy('กำลังโหลดรายการรายงาน', 'กำลังดึงข้อมูลเอกสารรายงานจากระบบ');

  runServer(
    'listReports',
    [{
      q: repState.q,
      page: repState.page,
      pageSize: repState.pageSize
    }],
    'กำลังโหลดรายการรายงาน',
    'กำลังประมวลผลข้อมูล',
    function (res) {
      endBusy();

      repState.loading = false;
      repState.rows = Array.isArray(res && res.rows) ? res.rows : [];
      repState.total = (res && typeof res.total === 'number') ? res.total : 0;

      renderReportsRows(repState.rows);
      renderPager();
      updateReportsUI();

      console.log(`Reports loaded: ${repState.rows.length} / ${repState.total}`);
    },
    function (err) {
      endBusy();

      repState.loading = false;
      console.error('loadReports error:', err);

      repState.rows = [];
      repState.total = 0;

      renderReportsRows([]);
      renderPager();
      updateReportsUI();
    }
  );
}

function getStatusBadge(statusText) {
  const s = String(statusText || '').trim();

  if (s === 'เสร็จสิ้น') {
    return { cls: 'badge badge-status-completed', icon: 'fa-circle-check', text: s };
  }
  if (s === 'กำลังดำเนินการ') {
    return { cls: 'badge badge-status-processing', icon: 'fa-spinner', text: s };
  }

     if (s === 'ดำเนินการภายนอก') {
    return { cls: 'badge badge-status-external', icon: 'fa-truck-fast', text: s };
  }

  if (s === 'รอดำเนินการ') {
    return { cls: 'badge badge-status-pending', icon: 'fa-hourglass-half', text: s };
  }
  if (s === 'ยกเลิก') {
    return { cls: 'badge badge-status-cancelled', icon: 'fa-circle-xmark', text: s };
  }
  if (s === 'พัสดุอยู่ระหว่างรอเบิก' || s === 'รอเบิก') {
    return { cls: 'badge badge-status-parts', icon: 'fa-box', text: s };
  }

  return { cls: 'badge badge-status-pending', icon: 'fa-circle-info', text: s || '-' };
}

function getPriorityBadge(priorityText) {
  const p = String(priorityText || '').trim();

  if (p === 'ฉุกเฉิน') {
    return { cls: 'badge badge-prio-critical', icon: 'fa-triangle-exclamation', text: p };
  }
  if (p === 'เร่งด่วน') {
    return { cls: 'badge badge-prio-urgent', icon: 'fa-bolt', text: p };
  }
  if (p === 'ปานกลาง') {
    return { cls: 'badge badge-prio-medium', icon: 'fa-gauge', text: p };
  }
  if (p === 'ต่ำ') {
    return { cls: 'badge badge-prio-low', icon: 'fa-leaf', text: p };
  }

  return { cls: 'badge badge-prio-low', icon: 'fa-minus', text: p || '-' };
}

  function initMobileRuntime() {

  };
