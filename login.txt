// ฟังก์ชัน navigation ที่ปรับปรุงแล้วสำหรับ IFRAME mode
function goToPage(page, newTab = false) {
    const targetUrl = window.baseUrl + '?page=' + page;
    console.log('🚀 Navigating to:', targetUrl, 'New tab:', newTab);
    
    try {
        if (newTab) {
            // เปิดใน tab ใหม่
            const newWindow = window.open(targetUrl, '_blank');
            if (!newWindow) {
                console.warn('⚠️ Popup blocked, using fallback');
                fallbackNavigation(targetUrl);
            }
        } else {
            // Navigation ใน tab เดียวกัน - ใช้หลายวิธี
            
            // Method 1: ลองใช้ window.top (สำหรับ IFRAME)
            try {
                if (window.top && window.top.location) {
                    window.top.location.href = targetUrl;
                    return;
                }
            } catch (e) {
                console.log('📝 window.top blocked, trying alternatives...');
            }
            
            // Method 2: ลองใช้ window.parent
            try {
                if (window.parent && window.parent.location) {
                    window.parent.location.href = targetUrl;
                    return;
                }
            } catch (e) {
                console.log('📝 window.parent blocked, trying alternatives...');
            }
            
            // Method 3: ใช้ window.location
            try {
                window.location.href = targetUrl;
                return;
            } catch (e) {
                console.log('📝 window.location blocked, using fallback...');
            }
            
            // Method 4: Fallback - ใช้ form submission
            fallbackNavigation(targetUrl);
        }
    } catch (error) {
        console.error('❌ Navigation error:', error);
        fallbackNavigation(targetUrl);
    }
}

// ฟังก์ชัน fallback สำหรับ navigation (สำคัญ: ใช้ form submit target='_top' เพื่อ bypass sandbox)
function fallbackNavigation(url) {
    console.log('🔄 Using fallback navigation to:', url);
    
    // สร้าง form สำหรับ navigation (วิธีที่แน่นอนที่สุด)
    const form = document.createElement('form');
    form.method = 'GET';
    form.action = url;
    form.target = '_top'; // สำคัญ: target="_top" สำหรับ IFRAME mode
    form.style.display = 'none';
    
    document.body.appendChild(form);
    form.submit();
}

// ฟังก์ชันสำหรับการนำทางโดยตรง (ใช้ใน handleLoginSuccess)
function goToPageDirectly(url) {
    // ลองหลายวิธี navigation
    const methods = [
        () => { if (window.top) window.top.location.replace(url); },
        () => { if (window.parent) window.parent.location.replace(url); },
        () => { window.location.replace(url); },
        () => { window.location.href = url; },
        () => { fallbackNavigation(url); }
    ];
    
    for (let i = 0; i < methods.length; i++) {
        try {
            methods[i]();
            console.log(`✅ Navigation method ${i + 1} succeeded`);
            return;
        } catch (e) {
            console.log(`📝 Navigation method ${i + 1} failed:`, e.message);
            continue;
        }
    }
}

    function handleLoginSuccess(result) {
    console.log('Login success data received from server:', result);

    // 1. เก็บข้อมูลผู้ใช้ลงใน localStorage
    if (result && result.user) {
        localStorage.setItem('userInfo', JSON.stringify(result.user));
    }

    // 2. แสดงข้อความแจ้งเตือนสั้นๆ
    showAlert('success', `เข้าสู่ระบบสำเร็จ! กำลังนำท่านไปยังหน้าหลัก...`);

    // 3. Redirect ด้วย fallback
    if (result && result.redirectUrl) {
        fallbackNavigation(result.redirectUrl);
    } else {
        // Fallback ถ้าไม่มี redirectUrl
        console.warn('⚠️ No redirectUrl, falling back to summary');
        fallbackNavigation(window.baseUrl + '?page=summary');
    }
}

function goBack() {
    console.log('Going back to main page...');

}

// ฟังก์ชัน handleLoginSubmit (ตรวจสอบว่าเหมือนเดิม)
document.getElementById('loginForm').addEventListener('submit', function(e) {
    e.preventDefault();
    const username = document.getElementById('username').value.trim();
    const password = document.getElementById('password').value.trim();
    if (!username || !password) {
        showAlert('error', 'กรุณากรอกชื่อผู้ใช้และรหัสผ่าน');
        return;
    }
    toggleLoading(true);
    google.script.run
        .withSuccessHandler(function(result) {
            // ไม่ต้องทำอะไรมากที่นี่ เพราะ handleLoginSuccess จะจัดการทุกอย่าง
            if (result && result.success) {
                handleLoginSuccess(result);
            } else {
                toggleLoading(false);
                showAlert('error', result.message || 'ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง');
                document.getElementById('password').value = '';
            }
        })
        .withFailureHandler(function(error) {
            toggleLoading(false);
            showAlert('error', 'เกิดข้อผิดพลาดในการเชื่อมต่อ: ' + error.message);
            console.error('Login error:', error);
        })
        .doLogin(username, password);
});

        // ฟังก์ชันตรวจสอบความปลอดภัยของข้อมูลที่กรอก
        function validateLoginInput(username, password) {
            // ตรวจสอบความยาวของชื่อผู้ใช้
            if (username.length < 3 || username.length > 50) {
                showAlert('error', 'ชื่อผู้ใช้ต้องมีความยาว 3-50 ตัวอักษร');
                return false;
            }
            
            // ตรวจสอบอักขระที่ไม่พึงประสงค์
            const dangerousChars = /[<>'";&()]/;
            if (dangerousChars.test(username) || dangerousChars.test(password)) {
                showAlert('error', 'ข้อมูลที่กรอกมีอักขระที่ไม่อนุญาต');
                return false;
            }
            
            // ตรวจสอบความยาวของรหัสผ่าน
            if (password.length < 4) {
                showAlert('error', 'รหัสผ่านต้องมีความยาวอย่างน้อย 4 ตัวอักษร');
                return false;
            }
            
            return true;
        }

        function checkExistingLogin() {
            const userInfo = localStorage.getItem('userInfo');
            if (userInfo) {
                try {
                    const user = JSON.parse(userInfo);
                    if (user && user.username) {
                        showAlert('info', `คุณยังเข้าสู่ระบบอยู่ในนาม: ${user.name || user.username}`);
                        document.getElementById('username').placeholder = 'คุณเข้าสู่ระบบอยู่แล้ว';
                        document.getElementById('password').placeholder = 'กรอกรหัสผ่านใหม่เพื่อเข้าสู่ระบบอีกครั้ง';
                    }
                } catch (e) {
                    // ล้างข้อมูลที่เสียหาย
                    localStorage.removeItem('userInfo');
                }
            }
        }

        function toggleLoading(show) {

                loginBtn.innerHTML = '<i class="fas fa-sign-in-alt"></i>เข้าสู่ระบบ';

        // ป้องกันการ copy/paste ที่อาจมีโค้ดอันตราย
        document.addEventListener('paste', function(e) {
            const clipboardData = e.clipboardData || window.clipboardData;
            const pastedData = clipboardData.getData('Text');
            
            // ตรวจสอบอักขระที่อันตราย
            const dangerousPattern = /<script|javascript:|onload=|onerror=/i;
            if (dangerousPattern.test(pastedData)) {
                e.preventDefault();
                showAlert('error', 'ไม่อนุญาตให้วางข้อมูลประเภทนี้');
            }
        });
